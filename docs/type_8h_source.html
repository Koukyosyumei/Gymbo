539  switch (symtype) {
540  case (SymType::SAdd): {
541  return left->grad(cvals, eps) + right->grad(cvals, eps);
542  }
543  case (SymType::SSub): {
544  return left->grad(cvals, eps) - right->grad(cvals, eps);
545  }
546  case (SymType::SMul): {
547  return left->grad(cvals, eps) * right->eval(cvals, eps) +
548  right->grad(cvals, eps) * left->eval(cvals, eps);
549  }
550  case (SymType::SCon): {
551  return Grad({});
552  }
553  case (SymType::SAny): {
554  std::unordered_map<int, float> tmp = {{var_idx, 1.0f}};
555  return Grad(tmp);
556  }
557  case (SymType::SEq): {
558  float lv = left->eval(cvals, eps);
559  float rv = right->eval(cvals, eps);
560  Grad lg = left->grad(cvals, eps);
561  Grad rg = right->grad(cvals, eps);
562  if (lv == rv) {
563  return Grad({});
564  } else if (lv > rv) {
565  return lg - rg;
566  } else {
567  return rg - lg;
568  }
569  }
570  case (SymType::SNot): {
571  return left->grad(cvals, eps) * (-1.0f);
572  }
573  case (SymType::SAnd): {
574  float lv = left->eval(cvals, eps);
575  float rv = right->eval(cvals, eps);
576  Grad res({});
577  if (lv > 0.0f) {
578  res = res + left->grad(cvals, eps);
579  }
580  if (rv > 0.0f) {
581  res = res + right->grad(cvals, eps);
582  }
583  return res;
584  }
585  case (SymType::SOr): {
586  float lv = left->eval(cvals, eps);
587  float rv = right->eval(cvals, eps);
588  if (lv > 0.0f && rv > 0.0f) {

590  right->grad(cvals, eps) * lv;
591  }
592  return Grad({});
593  }
594  case (SymType::SLt): {
595  return left->grad(cvals, eps) - right->grad(cvals, eps);
596  }
597  case (SymType::SLe): {
