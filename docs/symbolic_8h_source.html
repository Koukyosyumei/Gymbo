501  bool solve(bool is_target, int pc, SymState &state) {
502  bool is_sat = true;
503  std::string constraints_str = state.toString(false);
504 
505  std::unordered_map<int, float> params = {};
506  initialize_params(params, state, ignore_memory);
507 
508  bool is_unknown_path_constraint = true;
509 
510  if (constraints_cache.find(constraints_str) !=
511  constraints_cache.end()) {
512  is_sat = constraints_cache[constraints_str].first;
513  params = constraints_cache[constraints_str].second;
514  is_unknown_path_constraint = false;
515  } else {
516  call_smt_solver(is_sat, state, params, optimizer, max_num_trials,
517  ignore_memory, use_dpll);
518  if (is_sat) {
519  maxSAT--;
520  } else {
521  maxUNSAT--;
522  }
523  constraints_cache.emplace(constraints_str,
524  std::make_pair(is_sat, params));
525  }
526 
527  if (verbose_level >= 1) {
528  verbose_constraints(verbose_level, is_unknown_path_constraint,
529  is_target, is_sat, pc, constraints_str, state,
530  params);
531  }
532 
533  return is_sat;
534  }

547  Trace run(SymState &state, int maxDepth = 256) {
548  int pc = state.pc;
549  bool is_target = is_target_pc(target_pcs, pc);
550  bool is_sat = true;
551 
552  verbose_pre(verbose_level, pc, prog, state);
553 
554  if (state.path_constraints.size() != 0 && is_target) {
555  is_sat = solve(is_target, pc, state);
556  }
557  verbose_post(verbose_level);
558 
559  if ((prog[pc].instr == InstrType::Done) || (!is_sat)) {
560  return Trace(state, {});
561  } else if (explore_further(maxDepth, maxSAT, maxUNSAT)) {
562  Instr instr = prog[pc];
563  std::vector<SymState *> newStates;
564  symStep(&state, instr, newStates);
565  std::vector<Trace> children;
566  for (SymState *newState : newStates) {
567  Trace child = run(*newState, maxDepth - 1);
568  if (return_trace) {
569  children.push_back(child);
570  }
571  }
572  return Trace(state, children);
573  } else {
574  return Trace(state, {});
575  }
576  }
