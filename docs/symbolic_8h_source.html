178 inline void solve_constraints(bool &is_sat, bool is_target, int pc,
179  GDOptimizer &optimizer, SymState &state,
180  PathConstraintsTable &constraints_cache,
181  int &maxSAT, int &maxUNSAT, int max_num_trials,
182  bool ignore_memory, bool use_dpll,
183  int verbose_level) {
184  std::string constraints_str = state.toString(false);
185 
186  std::unordered_map<int, float> params = {};
187  initialize_params(params, state, ignore_memory);
188 
189  bool is_unknown_path_constraint = true;
190 
191  if (constraints_cache.find(constraints_str) != constraints_cache.end()) {
192  is_sat = constraints_cache[constraints_str].first;
193  params = constraints_cache[constraints_str].second;
194  is_unknown_path_constraint = false;

201  maxUNSAT--;
202  }

204  std::make_pair(is_sat, params));
205  }
206 
207  if (verbose_level >= 1) {
208  verbose_constraints(verbose_level, is_unknown_path_constraint,
209  is_target, is_sat, pc, constraints_str, state,
210  params);

241 inline Trace run_gymbo(Prog &prog, GDOptimizer &optimizer, SymState &state,
242  std::unordered_set<int> &target_pcs,
243  PathConstraintsTable &constraints_cache, int maxDepth,
244  int &maxSAT, int &maxUNSAT, int max_num_trials,
245  bool ignore_memory, bool use_dpll, int verbose_level,
246  bool return_trace = false) {
247  int pc = state.pc;
248  bool is_target = is_target_pc(target_pcs, pc);
249  bool is_sat = true;
250 
251  verbose_pre(verbose_level, pc, prog, state);
252  if (state.path_constraints.size() != 0 && is_target) {
253  solve_constraints(is_sat, is_target, pc, optimizer, state,
254  constraints_cache, maxSAT, maxUNSAT, max_num_trials,
255  ignore_memory, use_dpll, verbose_level);
256  }
257  verbose_post(verbose_level);
258 
259  if ((prog[pc].instr == InstrType::Done) || (!is_sat)) {
260  return Trace(state, {});
261  } else if (explore_further(maxDepth, maxSAT, maxUNSAT)) {
262  Instr instr = prog[pc];
263  std::vector<SymState *> newStates;
264  symStep(&state, instr, newStates);
265  std::vector<Trace> children;
266  for (SymState *newState : newStates) {
267  Trace child = run_gymbo(prog, optimizer, *newState, target_pcs,
268  constraints_cache, maxDepth - 1, maxSAT,
269  maxUNSAT, max_num_trials, ignore_memory,
270  use_dpll, verbose_level, return_trace);
271  if (return_trace) {
272  children.push_back(child);
273  }
274  }
275  return Trace(state, children);
276  } else {
277  return Trace(state, {});
278  }

482  break;
483  }
484  case InstrType::JmpIf: {

486  state->symbolic_stack.pop();

488  state->symbolic_stack.pop();
489  if (addr->symtype == SymType::SCon) {

491  SymState *false_state = state->copy();
492  true_state->pc += wordToInt(addr->word - 2);
493  true_state->path_constraints.emplace_back(*cond);
494  false_state->pc++;
495  false_state->path_constraints.emplace_back(
496  Sym(SymType::SNot, cond));
497  result.emplace_back(true_state);
498  result.emplace_back(false_state);
499  }
500  break;
501  }
502  case InstrType::Jmp: {
503  Sym *addr = state->symbolic_stack.back();
504  state->symbolic_stack.pop();
505  state->pc += wordToInt(addr->word);
506  result.emplace_back(state);
507  break;
508  }
509  case InstrType::Nop: {
510  state->pc++;
511  result.emplace_back(state);
512  break;
513  }
514  case InstrType::Done: {
515  break;
516  }
517  default:
518  fprintf(stderr, "Detect unsupported instruction\n");
519  }
