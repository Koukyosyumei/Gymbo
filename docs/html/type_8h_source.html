497  switch (symtype) {

499  return left->grad(cvals, eps) + right->grad(cvals, eps);
500  }
501  case (SymType::SSub): {
502  return left->grad(cvals, eps) - right->grad(cvals, eps);
503  }
504  case (SymType::SMul): {
505  return left->grad(cvals, eps) * right->eval(cvals, eps) +
506  right->grad(cvals, eps) * left->eval(cvals, eps);
507  }
508  case (SymType::SCon): {
509  return Grad({});
510  }
511  case (SymType::SAny): {
512  std::unordered_map<int, float> tmp = {{var_idx, 1.0f}};
513  return Grad(tmp);
514  }
515  case (SymType::SEq): {
516  float lv = left->eval(cvals, eps);
517  float rv = right->eval(cvals, eps);
518  Grad lg = left->grad(cvals, eps);
519  Grad rg = right->grad(cvals, eps);
520  if (lv == rv) {
521  return Grad({});
522  } else if (lv > rv) {
523  return lg - rg;
524  } else {
525  return rg - lg;
526  }
527  }
528  case (SymType::SNot): {
529  return left->grad(cvals, eps) * (-1.0f);
530  }
531  case (SymType::SAnd): {
532  float lv = left->eval(cvals, eps);
533  float rv = right->eval(cvals, eps);
534  Grad res({});
535  if (lv > 0.0f) {
536  res = res + left->grad(cvals, eps);
537  }
538  if (rv > 0.0f) {
539  res = res + right->grad(cvals, eps);
540  }
541  return res;
542  }
543  case (SymType::SOr): {
544  float lv = left->eval(cvals, eps);
545  float rv = right->eval(cvals, eps);
546  if (lv > 0.0f && rv > 0.0f) {

548  right->grad(cvals, eps) * lv;
549  }
550  return Grad({});
551  }
552  case (SymType::SLt): {
553  return left->grad(cvals, eps) - right->grad(cvals, eps);
554  }
