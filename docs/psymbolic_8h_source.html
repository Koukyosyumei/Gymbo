26  std::vector<std::vector<int>> &D, SymState &state) {
27  int n_path_constraints = state.path_constraints.size();
28  if (state.has_observed_p_cond) {
29  Sym *n_cond = &state.path_constraints[0];
30  Sym *d_cond = &state.path_constraints[0];
31  for (int i = 1; i < n_path_constraints - 1; i++) {
32  n_cond = new Sym(SymType::SAnd, n_cond, &state.path_constraints[i]);
33  d_cond = new Sym(SymType::SAnd, d_cond, &state.path_constraints[i]);
34  }
35  n_cond = new Sym(SymType::SAnd, n_cond,
36  &state.path_constraints[n_path_constraints - 1]);
37  state.p = state.p * SymProb(n_cond, d_cond);
38  } else {
39  Sym *n_cond;
40  if (n_path_constraints == 0) {
41  n_cond = new Sym(SymType::SCon, FloatToWord(0.0f));
42  } else {
43  n_cond = &state.path_constraints[0];
44  }
45  for (int i = 1; i < n_path_constraints; i++) {
46  n_cond = new Sym(SymType::SAnd, n_cond, &state.path_constraints[i]);
47  }
48  Sym *d_cond = new Sym(SymType::SCon, FloatToWord((float)D.size()));
49  state.p = SymProb(n_cond, d_cond);
50  state.has_observed_p_cond = true;
51  }

93  bool ignore_memory, bool use_dpll,
94  int verbose_level) {

96 
97  std::unordered_map<int, float> params = {};
98  initialize_params(params, state, ignore_memory);
99 
100  bool is_unknown_path_constraint = true;
101 
102  if (constraints_cache.find(constraints_str) != constraints_cache.end()) {
103  is_sat = constraints_cache[constraints_str].first;
104  params = constraints_cache[constraints_str].second;
105  is_unknown_path_constraint = false;
106  } else {
107  bool is_contain_prob_var = false;
108  std::unordered_set<int> unique_var_ids;
109  for (int i = 0; i < state.path_constraints.size(); i++) {
110  state.path_constraints[i].gather_var_ids(unique_var_ids);
111  }
112  for (int i : unique_var_ids) {
113  if (var2dist.find(i) != var2dist.end()) {
114  is_contain_prob_var = true;
115  break;
116  }
117  }
118 
119  if (is_contain_prob_var) {
120  // call probabilistic branch algorithm
121  pbranch(var2dist, D, state);
122  is_sat = true;
123  } else {
124  // solve deterministic path constraints
125  call_smt_solver(is_sat, state, params, optimizer, max_num_trials,
126  ignore_memory, use_dpll);
127  if (is_sat) {
128  maxSAT--;
129  } else {
130  maxUNSAT--;
131  }
132  }
133 
134  constraints_cache.emplace(constraints_str,
135  std::make_pair(is_sat, params));
136  }
137 
138  if (verbose_level >= 1) {
139  verbose_pconstraints(verbose_level, is_unknown_path_constraint,
140  is_target, is_sat, pc, constraints_str, state,

142  }

191 inline Trace run_pgymbo(Prog &prog,
192  std::unordered_map<int, DiscreteDist> &var2dist,
193  std::vector<std::vector<int>> &D,
194  GDOptimizer &optimizer, SymState &state,
195  std::unordered_set<int> &target_pcs,
196  PathConstraintsTable &constraints_cache,
197  ProbPathConstraintsTable &prob_constraints_table,
198  int maxDepth, int &maxSAT, int &maxUNSAT,
199  int max_num_trials, bool ignore_memory, bool use_dpll,
200  int verbose_level, bool return_trace = false) {
201  int pc = state.pc;
202  bool is_target = is_target_pc(target_pcs, pc);
203  bool is_sat = true;
204 
205  verbose_pre(verbose_level, pc, prog, state);
206  if (state.path_constraints.size() != 0 && is_target) {
207  solve_pconstraints(is_sat, is_target, pc, var2dist, D, optimizer, state,
208  constraints_cache, maxSAT, maxUNSAT, max_num_trials,
209  ignore_memory, use_dpll, verbose_level);
210  }
211  verbose_post(verbose_level);
212 
213  if (prog[pc].instr == InstrType::Done &&
214  state.path_constraints.size() > 0) {
215  update_prob_constraints_table(pc, state, prob_constraints_table);
216  }
217 
218  if ((prog[pc].instr == InstrType::Done) || (!is_sat)) {
219  return Trace(state, {});
220  } else if (explore_further(maxDepth, maxSAT, maxUNSAT)) {
221  Instr instr = prog[pc];
222  std::vector<SymState *> newStates;
223  symStep(&state, instr, newStates);
224  std::vector<Trace> children;
225  for (SymState *newState : newStates) {
226  Trace child = run_pgymbo(
227  prog, var2dist, D, optimizer, *newState, target_pcs,
228  constraints_cache, prob_constraints_table, maxDepth - 1, maxSAT,
229  maxUNSAT, max_num_trials, ignore_memory, use_dpll,
230  verbose_level, return_trace);
231  if (return_trace) {
232  children.push_back(child);
233  }
234  }
235  return Trace(state, children);
236  } else {
237  return Trace(state, {});
