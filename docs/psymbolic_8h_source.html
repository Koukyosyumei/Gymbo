22 inline void pbranch(SymState &state) {
23  int n_path_constraints = state.path_constraints.size();
24  if (state.has_observed_p_cond) {
25  Sym *n_cond = &state.path_constraints[0];
26  Sym *d_cond = &state.path_constraints[0];
27  for (int i = 1; i < n_path_constraints - 1; i++) {
28  n_cond = new Sym(SymType::SAnd, n_cond, &state.path_constraints[i]);
29  d_cond = new Sym(SymType::SAnd, d_cond, &state.path_constraints[i]);
30  }
31  n_cond = new Sym(SymType::SAnd, n_cond,
32  &state.path_constraints[n_path_constraints - 1]);
33  state.cond_p = new SymProb(n_cond, d_cond);
34  state.p = state.p->pmul(state.cond_p);
35  } else {
36  Sym *n_cond;
37  if (n_path_constraints == 0) {
38  n_cond = new Sym(SymType::SCon, FloatToWord(0.0f));
39  } else {
40  n_cond = &state.path_constraints[0];
41  }
42  for (int i = 1; i < n_path_constraints; i++) {
43  n_cond = new Sym(SymType::SAnd, n_cond, &state.path_constraints[i]);
44  }
45  Sym *d_cond = new Sym(SymType::SCon, FloatToWord(1.0f));
46  state.p = new SymProb(n_cond, d_cond);
47  state.cond_p = state.p;
48  state.has_observed_p_cond = true;
49  }
50 }

190  bool solve(bool is_target, int pc, SymState &state) {
191  std::string constraints_str = state.toString(false);
192 
193  std::unordered_map<int, float> params = {};
194  initialize_params(params, state, ignore_memory);
195 
196  bool is_sat = true;
197  bool is_unknown_path_constraint = true;
198 
199  if (constraints_cache.find(constraints_str) !=
200  constraints_cache.end()) {
201  is_sat = constraints_cache[constraints_str].first;
202  params = constraints_cache[constraints_str].second;
203  is_unknown_path_constraint = false;
204  } else {
205  bool is_contain_prob_var = false;
206  std::unordered_set<int> unique_var_ids;
207  for (int i = 0; i < state.path_constraints.size(); i++) {
208  state.path_constraints[i].gather_var_ids(unique_var_ids);
209  }
210  for (int i : unique_var_ids) {
211  if (random_vars.find(i) != random_vars.end()) {
212  is_contain_prob_var = true;
213  break;
214  }
215  }
216 
217  if (is_contain_prob_var) {
218  // call probabilistic branch algorithm
219  pbranch(state);
220  is_sat = true;
221  } else {
222  // solve deterministic path constraints
223  call_smt_solver(is_sat, state, params, optimizer,
224  max_num_trials, ignore_memory, use_dpll);
225  if (is_sat) {
226  maxSAT--;
227  } else {
228  maxUNSAT--;
229  }
230 
231  if (is_sat) {
232  state.p =
233  new SymProb(new Sym(SymType::SCon, FloatToWord(0.0f)),
234  new Sym(SymType::SCon, FloatToWord(0.0f)));
235  state.cond_p = state.p;
236  }
237  }
238 
239  constraints_cache.emplace(constraints_str,
240  std::make_pair(is_sat, params));
241  }
242 
243  if (verbose_level >= 1) {
244  verbose_pconstraints(verbose_level, is_unknown_path_constraint,
245  is_target, is_sat, pc, constraints_str, state,
246  params);
247  }
248 
249  return is_sat;
250  }

264  Trace run(SymState &state, int maxDepth = 256) {
265  int pc = state.pc;
266  bool is_target = is_target_pc(target_pcs, pc);
267  bool is_sat = true;
268 
269  verbose_pre(verbose_level, pc, prog, state);
270  if (state.path_constraints.size() != 0 && is_target) {
271  is_sat = solve(is_target, pc, state);
272  }
273  verbose_post(verbose_level);
274 
275  if (prog[pc].instr == InstrType::Done &&
276  state.path_constraints.size() > 0) {
277  update_prob_constraints_table(pc, state, prob_constraints_table);
278  }
279 
280  if ((prog[pc].instr == InstrType::Done) || (!is_sat)) {
281  return Trace(state, {});
282  } else if (explore_further(maxDepth, maxSAT, maxUNSAT)) {
283  Instr instr = prog[pc];
284  std::vector<SymState *> newStates;
285  symStep(&state, instr, newStates);
286  std::vector<Trace> children;
287  for (SymState *newState : newStates) {
288  Trace child = run(*newState, maxDepth - 1);
289  if (return_trace) {
290  children.push_back(child);
291  }
292  }
293  return Trace(state, children);
294  } else {
295  return Trace(state, {});
296  }
297  }
